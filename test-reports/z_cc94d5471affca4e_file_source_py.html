47 def fetch_data(self) -> Iterator[IPEntry]: 

48 """파일에서 IP 데이터 읽기""" 

49 file_path = self.config.settings["file_path"] 

50 file_format = self.config.settings.get("format", "auto") 

51 delimiter = self.config.settings.get("delimiter", None) 

52 ip_column = self.config.settings.get("ip_column", 0) 

53 category = self.config.settings.get("category", "blacklist") 

54 

55 try: 

56 # 파일 형식 자동 감지 

57 if file_format == "auto": 

58 file_format = self._detect_format(file_path) 

59 

60 # 형식별 파싱 

61 if file_format == "txt": 

62 yield from self._parse_text_file(file_path, category) 

63 elif file_format == "csv": 

64 yield from self._parse_csv_file( 

65 file_path, delimiter, ip_column, category 

66 ) 

67 elif file_format == "json": 

68 yield from self._parse_json_file(file_path, category) 

69 elif file_format == "log": 

70 yield from self._parse_log_file(file_path, category) 

71 else: 

72 raise ValueError(f"Unsupported file format: {file_format}") 

73 

74 except Exception as e: 

75 self.logger.error(f"Failed to parse file {file_path}: {e}")

78 def _detect_format(self, file_path: str) -> str: 

79 """파일 형식 자동 감지""" 

80 extension = Path(file_path).suffix.lower() 

81 

82 format_map = { 

83 ".txt": "txt", 

84 ".csv": "csv", 

85 ".json": "json", 

86 ".log": "log", 

87 ".tsv": "csv", 

88 } 

89 

90 return format_map.get(extension, "txt")

92 def _parse_text_file(self, file_path: str, category: str) -> Iterator[IPEntry]: 

93 """텍스트 파일 파싱 (한 줄에 하나의 IP)""" 

94 with open(file_path, "r", encoding="utf-8") as f: 

95 for line_num, line in enumerate(f, 1): 

96 line = line.strip() 

97 

98 # 빈 줄이나 주석 스킵 

99 if not line or line.startswith("#"): 

100 continue 

101 

102 # IP 주소 추출 

103 ip = self._extract_ip_from_line(line) 

104 if ip and self.is_valid_ip(ip): 

105 yield IPEntry( 

106 ip_address=ip, 

107 source_name=f"{self.source_name}:{Path(file_path).name}", 

108 category=category, 

109 confidence=1.0, 

110 detection_date=datetime.utcnow(), 

111 metadata={ 

112 "file_path": file_path, 

113 "line_number": line_num, 

114 "original_line": line, 

115 }, 

116 )

118 def _parse_csv_file( 

119 self, file_path: str, delimiter: str, ip_column: int, category: str 

120 ) -> Iterator[IPEntry]: 

121 """CSV 파일 파싱""" 

122 import csv 

123 

124 if delimiter is None: 

125 # 자동 구분자 감지 

126 with open(file_path, "r", encoding="utf-8") as f: 

127 sample = f.read(1024) 

128 sniffer = csv.Sniffer() 

129 delimiter = sniffer.sniff(sample).delimiter 

130 

131 with open(file_path, "r", encoding="utf-8") as f: 

132 reader = csv.reader(f, delimiter=delimiter) 

133 

134 for line_num, row in enumerate(reader, 1): 

135 if len(row) <= ip_column: 

136 continue 

137 

138 ip = row[ip_column].strip() 

139 if self.is_valid_ip(ip): 

140 # 추가 컬럼 정보를 메타데이터로 저장 

141 metadata = {"file_path": file_path, "line_number": line_num} 

142 

143 # 다른 컬럼들을 메타데이터에 추가 

144 for i, value in enumerate(row): 

145 if i != ip_column: 

146 metadata[f"column_{i}"] = value 

147 

148 yield IPEntry( 

149 ip_address=ip, 

150 source_name=f"{self.source_name}:{Path(file_path).name}", 

151 category=category, 

152 confidence=1.0, 

153 detection_date=datetime.utcnow(), 

154 metadata=metadata, 

155 )

158 """JSON 파일 파싱"""

160

163 

164 # JSON 구조 자동 감지 

165 if isinstance(data, list): 

166 # 단순 IP 리스트 

167 for ip in data: 

168 if isinstance(ip, str) and self.is_valid_ip(ip): 

169 yield IPEntry(

171 source_name=f"{self.source_name}:{Path(file_path).name}", 

172 category=category, 

173 confidence=1.0, 

174 detection_date=datetime.utcnow(), 

175 metadata={"file_path": file_path}, 

176 ) 

177 elif isinstance(data, dict): 

178 # 복잡한 JSON 구조 처리 

179 yield from self._parse_json_object(data, file_path, category)

181 def _parse_json_object( 

182 self, data: dict, file_path: str, category: str 

183 ) -> Iterator[IPEntry]: 

184 """복잡한 JSON 객체 파싱""" 

185 

186 def extract_ips(obj, path=""): 

187 if isinstance(obj, dict): 

188 for key, value in obj.items(): 

189 new_path = f"{path}.{key}" if path else key 

190 yield from extract_ips(value, new_path) 

191 elif isinstance(obj, list): 

192 for i, item in enumerate(obj): 

193 yield from extract_ips(item, f"{path}[{i}]") 

194 elif isinstance(obj, str) and self.is_valid_ip(obj): 

195 yield obj, path 

196 

197 for ip, json_path in extract_ips(data): 

198 yield IPEntry( 

199 ip_address=ip, 

200 source_name=f"{self.source_name}:{Path(file_path).name}", 

201 category=category, 

202 confidence=1.0, 

203 detection_date=datetime.utcnow(), 

204 metadata={"file_path": file_path, "json_path": json_path}, 

205 )

207 def _parse_log_file(self, file_path: str, category: str) -> Iterator[IPEntry]: 

208 """로그 파일에서 IP 추출""" 

209 # 일반적인 IP 패턴 

210 ip_pattern = re.compile(r"\b(?:\d{1,3}\.){3}\d{1,3}\b") 

211 

212 with open(file_path, "r", encoding="utf-8") as f: 

213 for line_num, line in enumerate(f, 1): 

214 # 모든 IP 주소 찾기 

215 ips = ip_pattern.findall(line) 

216 

217 for ip in ips: 

218 if self.is_valid_ip(ip): 

219 yield IPEntry( 

220 ip_address=ip, 

221 source_name=f"{self.source_name}:{Path(file_path).name}", 

222 category=category, 

223 confidence=0.8, # 로그에서 추출한 IP는 확신도 낮음 

224 detection_date=datetime.utcnow(), 

225 metadata={ 

226 "file_path": file_path, 

227 "line_number": line_num, 

228 "log_line": line.strip(), 

229 }, 

230 )

232 def _extract_ip_from_line(self, line: str) -> str: 

233 """라인에서 IP 주소 추출""" 

234 # 여러 형식 지원 

235 # 1. 순수 IP: "192.168.1.1" 

236 # 2. CIDR: "192.168.1.1/24" -> "192.168.1.1" 

237 # 3. 포트 포함: "192.168.1.1:80" -> "192.168.1.1" 

238 # 4. 공백으로 구분된 필드 

239 

240 # 공백으로 분할하여 첫 번째 필드 사용 

241 parts = line.split() 

242 if not parts: 

243 return None 

244 

245 ip_candidate = parts[0] 

246 

247 # CIDR 표기 처리 

248 if "/" in ip_candidate: 

249 ip_candidate = ip_candidate.split("/")[0] 

250 

251 # 포트 번호 제거 

252 if ":" in ip_candidate: 

253 ip_candidate = ip_candidate.split(":")[0] 

254 

255 return ip_candidate
