15 def __init__(self, session: requests.Session, base_url: str): 

16 self.session = session 

17 self.base_url = base_url 

18 

19 def get_file_content(self, project_id: str, file_path: str) -> str: 

20 """파일 내용 조회""" 

21 try: 

22 response = self.session.get( 

23 f"{self.base_url}/api/v1/projects/{project_id}/files/{file_path}" 

24 ) 

25 if response.status_code == 200: 

26 return response.json().get("content", "") 

27 return "" 

28 except Exception as e: 

29 print(f"❌ 파일 조회 실패 ({file_path}): {e}") 

30 return "" 

31 

32 def update_file(self, project_id: str, file_path: str, content: str) -> bool: 

33 """파일 내용 업데이트""" 

34 try: 

35 data = { 

36 "project_id": project_id, 

37 "file_path": file_path, 

38 "content": content, 

39 "commit_message": f"Auto-fix: {file_path}", 

40 } 

41 response = self.session.post(f"{self.base_url}/api/v1/files", json=data) 

42 if response.status_code in [200, 201]: 

43 print(f"✅ 파일 업데이트 완료: {file_path}") 

44 return True 

45 else: 

46 print(f"❌ 파일 업데이트 실패 ({file_path}): {response.status_code}") 

47 return False 

48 except Exception as e: 

49 print(f"❌ 파일 업데이트 실패 ({file_path}): {e}") 

50 return False

150 def backup_file(self, project_id: str, file_path: str) -> str: 

151 """파일 백업 (수정 전)""" 

152 try: 

153 content = self.get_file_content(project_id, file_path) 

154 if content: 

155 backup_path = f"{file_path}.backup" 

156 if self.update_file(project_id, backup_path, content): 

157 return backup_path 

158 return "" 

159 except Exception as e: 

160 print(f"❌ 파일 백업 실패: {e}") 

161 return ""

178 def analyze_log_patterns(self, log_content: str) -> Dict[str, Any]: 

179 """로그 패턴 분석""" 

180 import re 

181 

182 patterns = { 

183 "error_count": len(re.findall(r"(?i)error", log_content)), 

184 "warning_count": len(re.findall(r"(?i)warning", log_content)), 

185 "timeout_issues": len(re.findall(r"(?i)timeout|timed out", log_content)), 

186 "permission_issues": len( 

187 re.findall(r"(?i)permission denied|access denied", log_content) 

188 ), 

189 "network_issues": len( 

190 re.findall(r"(?i)network|dns|connection", log_content) 

191 ), 

192 } 

193 

194 return patterns

196 def generate_fix_report(self, project_id: str, fixes_applied: list) -> str: 

197 """수정 보고서 생성""" 

198 from datetime import datetime 

199 

200 timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S") 

201 

202 report = f"""# CI/CD Auto-Fix Report 

203 

204**Project**: {project_id} 

205**Timestamp**: {timestamp} 

206**Total Fixes Applied**: {len(fixes_applied)} 

207 

208## Applied Fixes: 

209 

210""" 

211 

212 for i, fix in enumerate(fixes_applied, 1): 

213 report += f"{i}. {fix.replace('_', ' ').title()}\n" 

214 

215 report += "\n---\n*Generated by Claude Code CI/CD Troubleshooter*\n" 

216 

217 return report
